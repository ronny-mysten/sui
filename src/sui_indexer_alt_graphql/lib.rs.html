71pub struct RpcArgs {

73    #[clap(long, default_value_t = Self::default().rpc_listen_address)]

81/// This type is responsible for the set-up and lifecycle of all services related to a GraphQL RPC
82/// service (the RPC service, online IDE, health checks etc). It is agnostic to the schema
83/// being served (which must be provided to `run`).
84pub struct RpcService<Q, M, S> {
85    /// Address to accept incoming RPC connections on.
86    rpc_listen_address: SocketAddr,
87
88    /// Whether to expose GraphiQL IDE on GET requests
89    with_ide: bool,
90
91    /// The version string to report with each response, as an HTTP header.
92    version: &'static str,
93
94    /// The axum router that will handle incoming requests.
95    router: Router,
96
97    /// The GraphQL schema this service will serve.
98    schema: SchemaBuilder<Q, M, S>,
99
100    /// Metrics for the RPC service.
101    metrics: Arc<RpcMetrics>,
102
103    /// Cancellation token controls lifecycle of all RPC-related services.
104    cancel: CancellationToken,
105}
106
107impl<Q, M, S> RpcService<Q, M, S>
108where
109    Q: ObjectType + 'static,
110    M: ObjectType + 'static,
111    S: SubscriptionType + 'static,
112{
113    pub fn new(
114        args: RpcArgs,
115        version: &'static str,
116        schema: SchemaBuilder<Q, M, S>,
117        registry: &Registry,
118        cancel: CancellationToken,
119    ) -> Self {
120        let RpcArgs {
121            rpc_listen_address,
122            no_ide,
123        } = args;
124
125        let metrics = RpcMetrics::new(registry);
126        let router = Router::new();
127
128        // The logging extension should be outermost so that it can surround all other extensions.
129        let schema = schema.extension(Logging(metrics.clone()));
130
131        Self {
132            rpc_listen_address,
133            with_ide: !no_ide,
134            version,
135            router,
136            schema,
137            metrics,
138            cancel,
139        }
140    }
141
142    /// Return a copy of the metrics.
143    pub fn metrics(&self) -> Arc<RpcMetrics> {
144        self.metrics.clone()
145    }
146
147    /// Add a handler to the axum router.
148    pub fn route(self, path: &str, router: MethodRouter) -> Self {
149        let router = self.router.route(path, router);
150        Self { router, ..self }
151    }
152
153    /// Add an extension as a layer to the axum router.
154    pub fn layer<T>(self, layer: T) -> Self
155    where
156        T: Clone + Send + Sync + 'static,
157    {
158        let router = self.router.layer(Extension(layer));
159        Self { router, ..self }
160    }
161
162    /// Add an extension to the GraphQL schema.
163    pub fn extension(self, ext: impl ExtensionFactory) -> Self {
164        let schema = self.schema.extension(ext);
165        Self { schema, ..self }
166    }
167
168    /// Add data to the GraphQL schema that can be accessed from the Context from any request.
169    pub fn data<D: Any + Send + Sync>(self, data: D) -> Self {
170        let schema = self.schema.data(data);
171        Self { schema, ..self }
172    }
173
174    /// Run the RPC service. This binds the listener and exposes handlers for the RPC service and IDE
175    /// (if enabled).
176    pub async fn run(self) -> anyhow::Result<JoinHandle<()>>
177    where
178        Q: ObjectType + 'static,
179        M: ObjectType + 'static,
180        S: SubscriptionType + 'static,
181    {
182        let Self {
183            rpc_listen_address,
184            with_ide,
185            version,
186            mut router,
187            schema,
188            metrics: _,
189            cancel,
190        } = self;
191
192        if with_ide {
193            info!("Starting GraphiQL IDE at 'http://{rpc_listen_address}/graphql'");
194            router = router.route("/graphql", get(graphiql));
195        } else {
196            info!("Skipping GraphiQL IDE setup");
197        }
198
199        router = router
200            .layer(Extension(schema.finish()))
201            .layer(axum::middleware::from_fn_with_state(
202                Version(version),
203                middleware::version::set_version,
204            ))
205            .layer(
206                cors::CorsLayer::new()
207                    .allow_methods([Method::POST])
208                    .allow_origin(cors::Any)
209                    .allow_headers(cors::Any),
210            );
211
212        info!("Starting GraphQL service on {rpc_listen_address}");
213        let listener = TcpListener::bind(rpc_listen_address)
214            .await
215            .context("Failed to bind GraphQL to listen address")?;
216
217        let service = axum::serve(
218            listener,
219            router.into_make_service_with_connect_info::<SocketAddr>(),
220        )
221        .with_graceful_shutdown({
222            let cancel = cancel.clone();
223            async move {
224                cancel.cancelled().await;
225                info!("Shutdown received, shutting down GraphQL service");
226            }
227        });
228
229        Ok(tokio::spawn(async move {
230            if let Err(e) = service.await.context("Failed to start GraphQL service") {
231                error!("Failed to start GraphQL service: {e:?}");
232                cancel.cancel();
233            }
234        }))
235    }
236}

242            no_ide: false,
243        }
244    }

247/// The GraphQL schema this service will serve, without any extensions or context added.
248pub fn schema() -> SchemaBuilder<Query, Mutation, EmptySubscription> {
249    Schema::build(Query::default(), Mutation, EmptySubscription)
250        .register_output_type::<IAddressable>()
251        .register_output_type::<IMoveDatatype>()
252        .register_output_type::<IMoveObject>()
253        .register_output_type::<IObject>()
254}

256/// Set-up and run the RPC service, using the provided arguments (expected to be extracted from the
257/// command-line). The service will continue to run until the cancellation token is triggered, and
258/// will signal cancellation on the token when it is shutting down.
259///
260/// Access to most reads is controlled by the `database_url` -- if it is `None`, those reads will
261/// not work. KV queries can optionally be served by a Bigtable instance or Ledger gRPC service
262/// via `kv_args`. If a Bigtable instance is configured, the `GOOGLE_APPLICATION_CREDENTIALS`
263/// environment variable must point to the credentials JSON file.
264///
265/// `version` is the version string reported in response headers by the service as part of every
266/// request.
267///
268/// The service may spin up auxiliary services (such as the system package task) to support itself,
269/// and will clean these up on shutdown as well.
270pub async fn start_rpc(
271    database_url: Option<Url>,
272    fullnode_args: FullnodeArgs,
273    db_args: DbArgs,
274    kv_args: args::KvArgs,
275    consistent_reader_args: ConsistentReaderArgs,
276    args: RpcArgs,
277    system_package_task_args: SystemPackageTaskArgs,
278    version: &'static str,
279    config: RpcConfig,
280    pg_pipelines: Vec<String>,
281    registry: &Registry,
282    cancel: CancellationToken,
283) -> anyhow::Result<JoinHandle<()>> {
284    let rpc = RpcService::new(args, version, schema(), registry, cancel.child_token());
285    let metrics = rpc.metrics();
286
287    // Create gRPC full node client wrapper
288    let fullnode_client = FullnodeClient::new(
289        Some("graphql_fullnode"),
290        fullnode_args,
291        registry,
292        cancel.child_token(),
293    )
294    .await?;
295
296    let pg_reader = PgReader::new(
297        Some("graphql_db"),
298        database_url.clone(),
299        db_args,
300        registry,
301        cancel.child_token(),
302    )
303    .await?;
304
305    let bigtable_reader = if let Some(instance_id) = kv_args.bigtable_instance.as_ref() {
306        let reader = BigtableReader::new(
307            instance_id.clone(),
308            "indexer-alt-graphql".to_owned(),
309            kv_args.bigtable_args(),
310            registry,
311        )
312        .await?;
313
314        Some(reader)
315    } else {
316        None
317    };
318
319    let ledger_grpc_reader = if let Some(ledger_grpc_url) = kv_args.ledger_grpc_url.as_ref() {
320        let reader = LedgerGrpcReader::new(ledger_grpc_url.clone(), kv_args.ledger_grpc_args())
321            .await
322            .context("Failed to create Ledger gRPC reader")?;
323        Some(reader)
324    } else {
325        None
326    };
327
328    let consistent_reader = ConsistentReader::new(
329        Some("graphql_consistent"),
330        consistent_reader_args,
331        registry,
332        cancel.child_token(),
333    )
334    .await?;
335
336    let pg_loader = Arc::new(pg_reader.as_data_loader());
337    let kv_loader = if let Some(reader) = bigtable_reader.as_ref() {
338        KvLoader::new_with_bigtable(Arc::new(reader.as_data_loader()))
339    } else if let Some(reader) = ledger_grpc_reader.as_ref() {
340        KvLoader::new_with_ledger_grpc(Arc::new(reader.as_data_loader()))
341    } else {
342        KvLoader::new_with_pg(pg_loader.clone())
343    };
344
345    let package_store = Arc::new(PackageCache::new(DbPackageStore::new(pg_loader.clone())));
346
347    let system_package_task = SystemPackageTask::new(
348        system_package_task_args,
349        pg_reader.clone(),
350        package_store.clone(),
351        cancel.child_token(),
352    );
353
354    // Fetch and cache the chain identifier from the database.
355    let chain_identifier = chain_identifier::task(
356        &pg_reader,
357        config.watermark.watermark_polling_interval,
358        cancel.child_token(),
359    )
360    .await?;
361
362    let watermark_task = WatermarkTask::new(
363        config.watermark,
364        pg_pipelines,
365        pg_reader.clone(),
366        bigtable_reader,
367        ledger_grpc_reader,
368        consistent_reader.clone(),
369        metrics.clone(),
370        cancel.child_token(),
371    );
372
373    let rpc = rpc
374        .route("/graphql", post(graphql))
375        .route("/graphql/health", get(health::check))
376        .layer(watermark_task.watermarks())
377        .layer(config.health)
378        .layer(DbProbe(database_url))
379        .extension(Timeout::new(config.limits.timeouts()))
380        .extension(QueryLimitsChecker::new(
381            config.limits.query_limits(),
382            metrics,
383        ))
384        .data(config.limits.pagination())
385        .data(config.limits)
386        .data(config.name_service)
387        .data(config.zklogin)
388        .data(chain_identifier)
389        .data(pg_reader)
390        .data(consistent_reader)
391        .data(pg_loader)
392        .data(kv_loader)
393        .data(package_store)
394        .data(fullnode_client);
395
396    let h_rpc = rpc.run().await?;
397    let h_system_package_task = system_package_task.run();
398    let h_watermark = watermark_task.run();
399
400    Ok(tokio::spawn(async move {
401        let _ = h_rpc.await;
402        cancel.cancel();
403        let _ = h_system_package_task.await;
404        let _ = h_watermark.await;
405    }))
406}

408/// Handler for RPC requests (POST requests making GraphQL queries).
409async fn graphql(
410    ConnectInfo(addr): ConnectInfo<SocketAddr>,
411    Extension(schema): Extension<Schema<Query, Mutation, EmptySubscription>>,
412    Extension(watermark): Extension<WatermarksLock>,
413    TypedHeader(content_length): TypedHeader<ContentLength>,
414    show_usage: Option<TypedHeader<ShowUsage>>,
415    request: GraphQLRequest,
416) -> GraphQLResponse {
417    let mut request = request
418        .into_inner()
419        .data(content_length)
420        .data(Session::new(addr))
421        .data(watermark.read().await.clone());
422
423    if let Some(TypedHeader(show_usage)) = show_usage {
424        request = request.data(show_usage);
425    }
426
427    schema.execute(request).await.into()
428}

430/// Handler for GET requests for the online IDE. GraphQL requests are forwarded to the POST handler
431/// at the same path.
432async fn graphiql(path: MatchedPath) -> Html<String> {
433    Html(GraphiQLSource::build().endpoint(path.as_str()).finish())
434}
