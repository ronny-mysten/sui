35    pub fn new(config_path: &Path) -> Result<Self, anyhow::Error> {
36        let config: SuiClientConfig = PersistedConfig::read(config_path).map_err(|err| {
37            anyhow!(
38                "Cannot open wallet config file at {:?}. Err: {err}",
39                config_path
40            )
41        })?;
42
43        let config = config.persisted(config_path);
44        let context = Self {
45            config,
46            request_timeout: None,
47            client: Default::default(),
48            max_concurrent_requests: None,
49            env_override: None,
50        };
51        Ok(context)
52    }

94    pub fn get_identity_address(
95        &mut self,
96        input: Option<KeyIdentity>,
97    ) -> Result<SuiAddress, anyhow::Error> {
98        if let Some(key_identity) = input {
99            if let Ok(address) = self.config.keystore.get_by_identity(&key_identity) {
100                return Ok(address);
101            }
102            if let Some(address) = self
103                .config
104                .external_keys
105                .as_ref()
106                .and_then(|external_keys| external_keys.get_by_identity(&key_identity).ok())
107            {
108                return Ok(address);
109            }
110
111            Err(anyhow!(
112                "No address found for the provided key identity: {key_identity}"
113            ))
114        } else {
115            self.active_address()
116        }
117    }

119    pub async fn get_client(&self) -> Result<SuiClient, anyhow::Error> {
120        let read = self.client.read().await;
121
122        Ok(if let Some(client) = read.as_ref() {
123            client.clone()
124        } else {
125            drop(read);
126            let client = self
127                .get_active_env()?
128                .create_rpc_client(self.request_timeout, self.max_concurrent_requests)
129                .await?;
130            self.client.write().await.insert(client).clone()
131        })
132    }

135        if self.env_override.is_some() {
136            self.config.get_env(&self.env_override).ok_or_else(|| {
137                anyhow!(
138                    "Environment configuration not found for env [{}]",
139                    self.env_override.as_deref().unwrap_or("None")
140                )
141            })
142        } else {
143            self.config.get_active_env()
144        }
145    }

147    // TODO: Ger rid of mut
148    pub fn active_address(&mut self) -> Result<SuiAddress, anyhow::Error> {
149        if self.config.keystore.entries().is_empty() {
150            return Err(anyhow!(
151                "No managed addresses. Create new address with `new-address` command."
152            ));
153        }
154
155        // Ok to unwrap because we checked that config addresses not empty
156        // Set it if not exists
157        self.config.active_address = Some(
158            self.config
159                .active_address
160                .unwrap_or(*self.config.keystore.addresses().first().unwrap()),
161        );
162
163        Ok(self.config.active_address.unwrap())
164    }

177    /// Get the latest full object reference given a object id
178    pub async fn get_full_object_ref(
179        &self,
180        object_id: ObjectID,
181    ) -> Result<FullObjectRef, anyhow::Error> {
182        let client = self.get_client().await?;
183        let object = client
184            .read_api()
185            .get_object_with_options(object_id, SuiObjectDataOptions::new().with_owner())
186            .await?
187            .into_object()?;
188        let object_ref = object.object_ref();
189        let owner = object
190            .owner
191            .expect("Owner should be present if `with_owner` is set");
192        Ok(FullObjectRef::from_object_ref_and_owner(object_ref, &owner))
193    }

195    /// Get all the gas objects (and conveniently, gas amounts) for the address
196    pub async fn gas_objects(
197        &self,
198        address: SuiAddress,
199    ) -> Result<Vec<(u64, SuiObjectData)>, anyhow::Error> {
200        let client = self.get_client().await?;
201
202        let mut objects: Vec<SuiObjectResponse> = Vec::new();
203        let mut cursor = None;
204        loop {
205            let response = client
206                .read_api()
207                .get_owned_objects(
208                    address,
209                    Some(SuiObjectResponseQuery::new(
210                        Some(SuiObjectDataFilter::StructType(GasCoin::type_())),
211                        Some(SuiObjectDataOptions::full_content()),
212                    )),
213                    cursor,
214                    None,
215                )
216                .await?;
217
218            objects.extend(response.data);
219
220            if response.has_next_page {
221                cursor = response.next_cursor;
222            } else {
223                break;
224            }
225        }
226
227        // TODO: We should ideally fetch the objects from local cache
228        let mut values_objects = Vec::new();
229
230        for object in objects {
231            let o = object.data;
232            if let Some(o) = o {
233                let gas_coin = GasCoin::try_from(&o)?;
234                values_objects.push((gas_coin.value(), o.clone()));
235            }
236        }
237
238        Ok(values_objects)
239    }

241    pub async fn get_object_owner(&self, id: &ObjectID) -> Result<SuiAddress, anyhow::Error> {
242        let client = self.get_client().await?;
243        let object = client
244            .read_api()
245            .get_object_with_options(*id, SuiObjectDataOptions::new().with_owner())
246            .await?
247            .into_object()?;
248        Ok(object
249            .owner
250            .ok_or_else(|| anyhow!("Owner field is None"))?
251            .get_owner_address()?)
252    }

265    /// Infer the sender of a transaction based on the gas objects provided. If no gas objects are
266    /// provided, assume the active address is the sender.
267    pub async fn infer_sender(&mut self, gas: &[ObjectID]) -> Result<SuiAddress, anyhow::Error> {
268        if gas.is_empty() {
269            return self.active_address();
270        }
271
272        // Find the owners of all supplied object IDs
273        let owners = future::try_join_all(gas.iter().map(|id| self.get_object_owner(id))).await?;
274
275        // SAFETY `gas` is non-empty.
276        let owner = owners.first().copied().unwrap();
277
278        ensure!(
279            owners.iter().all(|o| o == &owner),
280            "Cannot infer sender, not all gas objects have the same owner."
281        );
282
283        Ok(owner)
284    }

286    /// Find a gas object which fits the budget
287    pub async fn gas_for_owner_budget(
288        &self,
289        address: SuiAddress,
290        budget: u64,
291        forbidden_gas_objects: BTreeSet<ObjectID>,
292    ) -> Result<(u64, SuiObjectData), anyhow::Error> {
293        for o in self.gas_objects(address).await? {
294            if o.0 >= budget && !forbidden_gas_objects.contains(&o.1.object_id) {
295                return Ok((o.0, o.1));
296            }
297        }
298        Err(anyhow!(
299            "No non-argument gas objects found for this address with value >= budget {budget}. Run sui client gas to check for gas objects."
300        ))
301    }

310    pub async fn get_gas_objects_owned_by_address(
311        &self,
312        address: SuiAddress,
313        limit: Option<usize>,
314    ) -> anyhow::Result<Vec<ObjectRef>> {
315        let client = self.get_client().await?;
316        let results: Vec<_> = client
317            .read_api()
318            .get_owned_objects(
319                address,
320                Some(SuiObjectResponseQuery::new(
321                    Some(SuiObjectDataFilter::StructType(GasCoin::type_())),
322                    Some(SuiObjectDataOptions::full_content()),
323                )),
324                None,
325                limit,
326            )
327            .await?
328            .data
329            .into_iter()
330            .filter_map(|r| r.data.map(|o| o.object_ref()))
331            .collect();
332        Ok(results)
333    }

366    /// Returns all the account addresses managed by the wallet and their owned gas objects.
367    pub async fn get_all_accounts_and_gas_objects(
368        &self,
369    ) -> anyhow::Result<Vec<(SuiAddress, Vec<ObjectRef>)>> {
370        let mut result = vec![];
371        for address in self.get_addresses() {
372            let objects = self
373                .gas_objects(address)
374                .await?
375                .into_iter()
376                .map(|(_, o)| o.object_ref())
377                .collect();
378            result.push((address, objects));
379        }
380        Ok(result)

394    pub fn get_keystore_by_identity(
395        &self,
396        key_identity: &KeyIdentity,
397    ) -> Result<&Keystore, anyhow::Error> {
398        if self.config.keystore.get_by_identity(key_identity).is_ok() {
399            return Ok(&self.config.keystore);
400        }
401
402        if let Some(external_keys) = self.config.external_keys.as_ref() {
403            if external_keys.get_by_identity(key_identity).is_ok() {
404                return Ok(external_keys);
405            }
406        }
407
408        Err(anyhow!(
409            "No keystore found for the provided key identity: {key_identity}"
410        ))
411    }

432    pub async fn sign_secure(
433        &self,
434        key_identity: &KeyIdentity,
435        data: &TransactionData,
436        intent: Intent,
437    ) -> Result<Signature, anyhow::Error> {
438        let keystore = self.get_keystore_by_identity(key_identity)?;
439        let sig = keystore.sign_secure(&data.sender(), data, intent).await?;

443    /// Sign a transaction with a key currently managed by the WalletContext
444    pub async fn sign_transaction(&self, data: &TransactionData) -> Transaction {
445        let sig = self
446            .config
447            .keystore
448            .sign_secure(&data.sender(), data, Intent::sui_transaction())
449            .await
450            .unwrap();
451        // TODO: To support sponsored transaction, we should also look at the gas owner.
452        Transaction::from_data(data.clone(), vec![sig])
453    }

455    /// Execute a transaction and wait for it to be locally executed on the fullnode.
456    /// Also expects the effects status to be ExecutionStatus::Success.
457    pub async fn execute_transaction_must_succeed(
458        &self,
459        tx: Transaction,
460    ) -> SuiTransactionBlockResponse {
461        tracing::debug!("Executing transaction: {:?}", tx);
462        let response = self.execute_transaction_may_fail(tx).await.unwrap();
463        assert!(
464            response.status_ok().unwrap(),
465            "Transaction failed: {:?}",
466            response
467        );
468        response
469    }

471    /// Execute a transaction and wait for it to be locally executed on the fullnode.
472    /// The transaction execution is not guaranteed to succeed and may fail. This is usually only
473    /// needed in non-test environment or the caller is explicitly testing some failure behavior.
474    pub async fn execute_transaction_may_fail(
475        &self,
476        tx: Transaction,
477    ) -> anyhow::Result<SuiTransactionBlockResponse> {
478        let client = self.get_client().await?;
479        Ok(client
480            .quorum_driver_api()
481            .execute_transaction_block(
482                tx,
483                SuiTransactionBlockResponseOptions::new()
484                    .with_effects()
485                    .with_input()
486                    .with_events()
487                    .with_object_changes()
488                    .with_balance_changes(),
489                Some(sui_types::quorum_driver_types::ExecuteTransactionRequestType::WaitForLocalExecution),
490            )
491            .await?)
492    }
