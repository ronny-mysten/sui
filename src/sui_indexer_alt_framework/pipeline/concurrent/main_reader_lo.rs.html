24/// Starts a task for a tasked pipeline to track the main reader lo. The existence of
25/// `reader_interval` indicates whether the indexer was tasked, necessitating this task, or not.
26pub(super) fn track_main_reader_lo<H: Handler + 'static>(
27    reader_lo: Arc<SetOnce<AtomicU64>>,
28    reader_interval: Option<Duration>,
29    cancel: CancellationToken,
30    store: H::Store,
31) -> JoinHandle<()> {
32    tokio::spawn(async move {
33        let Some(reader_interval) = reader_interval else {
34            info!(
35                pipeline = H::NAME,
36                "Not a tasked indexer, skipping main reader lo task"
37            );
38            reader_lo.set(AtomicU64::new(0)).ok();
39            return;
40        };
41
42        let mut reader_interval = interval(reader_interval);
43
44        // If we miss ticks, skip them to ensure we have the latest watermark.
45        reader_interval.set_missed_tick_behavior(MissedTickBehavior::Skip);
46
47        loop {
48            tokio::select! {
49                _ = cancel.cancelled() => {
50                    info!(pipeline = H::NAME, "Shutdown received");
51                    break;
52                }
53
54                _ = reader_interval.tick() => {
55                    match store.connect().await {
56                        Ok(mut conn) => {
57                            match conn.reader_watermark(H::NAME).await {
58                                Ok(watermark_opt) => {
59                                    // If the reader watermark is not present (either because the
60                                    // watermark entry does not exist, or the reader watermark is
61                                    // not set), we assume that pruning is not enabled, and
62                                    // checkpoints >= 0 are valid.
63                                    let update = watermark_opt.map_or(0, |wm| wm.reader_lo);
64
65                                    let current = reader_lo.get();
66
67                                    if let Some(can_update) = current {
68                                        can_update.store(update, Ordering::Relaxed);
69                                    } else {
70                                        reader_lo.set(AtomicU64::new(update)).ok();
71                                    }
72                                }
73                                Err(e) => {
74                                    warn!(pipeline = H::NAME, "Failed to get reader watermark: {e}");
75                                }
76                            }
77                        }
78                        Err(e) => {
79                            warn!(pipeline = H::NAME, "Failed to connect to store: {e}");
80                        }
81                    }
82                }
83            }
84        }
85    })
86}
