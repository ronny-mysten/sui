18pub fn transaction(
19    meter: &mut TranslationMeter<'_, '_>,
20    env: &Env,
21    tx_context: &TxContext,
22    pt: P::ProgrammableTransaction,
23) -> Result<L::Transaction, ExecutionError> {
24    metering::pre_translation::meter(meter, &pt)?;
25    let P::ProgrammableTransaction { inputs, commands } = pt;
26    let inputs = inputs
27        .into_iter()
28        .map(|arg| input(env, tx_context, arg))
29        .collect::<Result<Vec<_>, _>>()?;
30    let commands = commands
31        .into_iter()
32        .enumerate()
33        .map(|(idx, cmd)| command(env, cmd).map_err(|e| e.with_command_index(idx)))
34        .collect::<Result<Vec<_>, _>>()?;
35    let loaded_tx = L::Transaction { inputs, commands };
36    metering::loading::meter(meter, &loaded_tx)?;
37    Ok(loaded_tx)
38}

40fn input(
41    env: &Env,
42    tx_context: &TxContext,
43    arg: CallArg,
44) -> Result<(L::InputArg, L::InputType), ExecutionError> {
45    Ok(match arg {
46        CallArg::Pure(bytes) => (L::InputArg::Pure(bytes), L::InputType::Bytes),
47        CallArg::Object(ObjectArg::Receiving(oref)) => {
48            (L::InputArg::Receiving(oref), L::InputType::Bytes)
49        }
50        CallArg::Object(ObjectArg::ImmOrOwnedObject(oref)) => {
51            let id = &oref.0;
52            let obj = env.read_object(id)?;
53            let Some(ty) = obj.type_() else {
54                invariant_violation!("Object {:?} has does not have a Move type", id);
55            };
56            let tag: StructTag = ty.clone().into();
57            let ty = env.load_type_from_struct(&tag)?;
58            let arg = match obj.owner {
59                Owner::AddressOwner(_) => L::ObjectArg::OwnedObject(oref),
60                Owner::Immutable => L::ObjectArg::ImmObject(oref),
61                Owner::ObjectOwner(_)
62                | Owner::Shared { .. }
63                | Owner::ConsensusAddressOwner { .. } => {
64                    invariant_violation!("Unexpected owner for ImmOrOwnedObject: {:?}", obj.owner);
65                }
66            };
67            (L::InputArg::Object(arg), L::InputType::Fixed(ty))
68        }
69        CallArg::Object(ObjectArg::SharedObject {
70            id,
71            initial_shared_version,
72            mutability,
73        }) => {
74            let obj = env.read_object(&id)?;
75            let Some(ty) = obj.type_() else {
76                invariant_violation!("Object {:?} has does not have a Move type", id);
77            };
78            let tag: StructTag = ty.clone().into();
79            let ty = env.load_type_from_struct(&tag)?;
80            let kind = match obj.owner {
81                Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
82                    invariant_violation!("Unexpected owner for SharedObject: {:?}", obj.owner)
83                }
84                Owner::Shared { .. } => L::SharedObjectKind::Legacy,
85                Owner::ConsensusAddressOwner { .. } => L::SharedObjectKind::Party,
86            };
87            (
88                L::InputArg::Object(L::ObjectArg::SharedObject {
89                    id,
90                    initial_shared_version,
91                    mutability: object_mutability(mutability),
92                    kind,
93                }),
94                L::InputType::Fixed(ty),
95            )
96        }
97        CallArg::FundsWithdrawal(f) => {
98            let FundsWithdrawalArg {
99                reservation,
100                type_arg,
101                withdraw_from,
102            } = f;
103            let amount = match reservation {
104                P::Reservation::EntireBalance => {
105                    invariant_violation!("Entire balance reservation amount is not yet supported")
106                }
107                P::Reservation::MaxAmountU64(u) => U256::from(u),
108                // TODO when types other than u64 are supported, we must check that this is a
109                // valid amount for the type
110            };
111            let funds_ty = match type_arg {
112                P::WithdrawalTypeArg::Balance(inner) => {
113                    let inner = env.load_type_input(0, inner)?;
114                    env.balance_type(inner)?
115                }
116            };
117            let ty = env.withdrawal_type(funds_ty.clone())?;
118            let owner: AccountAddress = match withdraw_from {
119                P::WithdrawFrom::Sender => tx_context.sender().into(),
120                P::WithdrawFrom::Sponsor => tx_context
121                    .sponsor()
122                    .ok_or_else(|| {
123                        make_invariant_violation!(
124                            "A sponsor withdrawal requires a sponsor and should have been \
125                            checked at signing"
126                        )
127                    })?
128                    .into(),
129            };
130            (
131                L::InputArg::FundsWithdrawal(L::FundsWithdrawalArg {
132                    amount,
133                    ty: ty.clone(),
134                    owner,
135                }),
136                L::InputType::Fixed(ty),
137            )
138        }
139    })
140}

142fn object_mutability(mutability: SharedObjectMutability) -> L::ObjectMutability {
143    match mutability {
144        SharedObjectMutability::Mutable => L::ObjectMutability::Mutable,
145        SharedObjectMutability::NonExclusiveWrite => L::ObjectMutability::NonExclusiveWrite,
146        SharedObjectMutability::Immutable => L::ObjectMutability::Immutable,
147    }
148}

150fn command(env: &Env, command: P::Command) -> Result<L::Command, ExecutionError> {
151    Ok(match command {
152        P::Command::MoveCall(pmc) => {
153            let resolved_linkage = env
154                .linkage_analysis
155                .compute_call_linkage(&pmc, env.linkable_store)?;
156            let P::ProgrammableMoveCall {
157                package,
158                module,
159                function: name,
160                type_arguments: ptype_arguments,
161                arguments,
162            } = *pmc;
163            let linkage = RootedLinkage::new(*package, resolved_linkage);
164            let type_arguments = ptype_arguments
165                .into_iter()
166                .enumerate()
167                .map(|(idx, ty)| env.load_type_input(idx, ty))
168                .collect::<Result<Vec<_>, _>>()?;
169            let function = env.load_function(package, module, name, type_arguments, linkage)?;
170            L::Command::MoveCall(Box::new(L::MoveCall {
171                function,
172                arguments,
173            }))
174        }
175        P::Command::MakeMoveVec(ptype_argument, arguments) => {
176            let type_argument = ptype_argument
177                .map(|ty| env.load_type_input(0, ty))
178                .transpose()?;
179            L::Command::MakeMoveVec(type_argument, arguments)
180        }
181        P::Command::TransferObjects(objects, address) => {
182            L::Command::TransferObjects(objects, address)
183        }
184        P::Command::SplitCoins(coin, amounts) => L::Command::SplitCoins(coin, amounts),
185        P::Command::MergeCoins(target, coins) => L::Command::MergeCoins(target, coins),
186        P::Command::Publish(items, object_ids) => {
187            let resolved_linkage = env
188                .linkage_analysis
189                .compute_publication_linkage(&object_ids, env.linkable_store)?;
190            L::Command::Publish(items, object_ids, resolved_linkage)
191        }
192        P::Command::Upgrade(items, object_ids, object_id, argument) => {
193            let resolved_linkage = env
194                .linkage_analysis
195                .compute_publication_linkage(&object_ids, env.linkable_store)?;
196            L::Command::Upgrade(items, object_ids, object_id, argument, resolved_linkage)
197        }
198    })
199}
